    <script>
        let gameState = {
            language: 'en',
            callMode: 'evenOdd',
            marbles: 100,
            wins: 0,
            bonusPoints: 0,
            currentRound: 1,
            currentMode: null,
            playerRole: 'guesser',
            hiddenMarbles: 0,
            selectedMarbles: new Set(),
            opponentGuess: null,
            opponentMarbles: 100,
            opponentName: 'AI',
            soundEnabled: true,
            adWatchesRemaining: 3,
            lastLoginDate: null,
            playTimeSeconds: 0,
            lastPlaytimeReward: 0
        };
        
        let playtimeInterval = null;
        
        function initGame() {
            loadGameData();
            processAutoRewards();
            startPlaytimeTracker();
            updateUI();
        }
        
        function processAutoRewards() {
            const today = new Date().toDateString();
            if (gameState.lastLoginDate !== today) {
                gameState.bonusPoints += 5;
                gameState.lastLoginDate = today;
                showNotification('🎁 Daily reward! +5 bonus points');
                saveGameData();
            }
        }
        
        function startPlaytimeTracker() {
            if (playtimeInterval) clearInterval(playtimeInterval);
            playtimeInterval = setInterval(() => {
                gameState.playTimeSeconds++;
                const timeSinceLastReward = gameState.playTimeSeconds - (gameState.lastPlaytimeReward * 1800);
                if (timeSinceLastReward >= 1800) {
                    gameState.bonusPoints += 10;
                    gameState.lastPlaytimeReward++;
                    showNotification('⏱️ Playtime reward! +10 bonus points');
                    updateUI();
                    saveGameData();
                }
                saveGameData();
            }, 1000);
        }
        
        function setCallMode(mode) {
            gameState.callMode = mode;
            document.getElementById('evenOddMode').classList.toggle('active', mode === 'evenOdd');
            document.getElementById('kaliJhotaMode').classList.toggle('active', mode === 'kaliJhota');
            saveGameData();
        }
        
        function showModeSelection() {
            document.getElementById('homeScreen').classList.add('hidden');
            document.getElementById('modeSelectionScreen').classList.add('active');
            document.getElementById('backBtn').classList.add('show');
        }
        
        function goBack() {
            const modeScreen = document.getElementById('modeSelectionScreen');
            const gameArea = document.getElementById('gameArea');
            
            if (modeScreen.classList.contains('active')) {
                modeScreen.classList.remove('active');
                document.getElementById('homeScreen').classList.remove('hidden');
                document.getElementById('backBtn').classList.remove('show');
            } else if (!gameArea.classList.contains('hidden')) {
                gameArea.classList.add('hidden');
                document.getElementById('homeScreen').classList.remove('hidden');
                document.getElementById('backBtn').classList.remove('show');
                resetGame();
            }
        }
        
        function startGame(mode) {
            if (mode === 'tournament' && gameState.marbles < 2500) {
                showNotification('You need 2500 marbles for tournament!');
                return;
            }
            
            gameState.currentMode = mode;
            gameState.currentRound = 1;
            gameState.playerRole = 'guesser';
            gameState.opponentMarbles = 100;
            
            const avatars = {ai: '🤖', random: '👤', challenge: '👥', tournament: '🏆'};
            const names = {ai: 'AI', random: 'Player' + Math.floor(Math.random() * 9999), challenge: 'Friend', tournament: 'Champion'};
            
            document.getElementById('player2Avatar').textContent = avatars[mode];
            gameState.opponentName = names[mode];
            
            document.getElementById('modeSelectionScreen').classList.remove('active');
            document.getElementById('gameArea').classList.remove('hidden');
            document.getElementById('backBtn').classList.add('show');
            
            setupRound();
        }
        
        function setupRound() {
            updateGameUI();
            document.getElementById('player1Box').classList.remove('active');
            document.getElementById('player2Box').classList.remove('active');
            
            if (gameState.playerRole === 'guesser') {
                document.getElementById('player1Box').classList.add('active');
                document.getElementById('player1Role').textContent = 'Guesser';
                document.getElementById('player2Role').textContent = 'Hider';
                
                document.getElementById('guessingSection').classList.remove('hidden');
                document.getElementById('marbleSelectionSection').classList.add('hidden');
                document.getElementById('revealSection').classList.add('hidden');
                document.getElementById('resultDisplay').classList.add('hidden');
                
                document.getElementById('fistDisplay').classList.remove('hidden');
                document.getElementById('fistContainer').textContent = '✊';
                document.getElementById('fistContainer').className = 'fist-container fist-closed';
                document.getElementById('revealedMarbles').classList.add('hidden');
                document.getElementById('fistLabel').textContent = gameState.opponentName + ' is hiding marbles...';
                
                setupGuessButtons();
            } else {
                document.getElementById('player2Box').classList.add('active');
                document.getElementById('player1Role').textContent = 'Hider';
                document.getElementById('player2Role').textContent = 'Guesser';
                
                document.getElementById('guessingSection').classList.add('hidden');
                document.getElementById('marbleSelectionSection').classList.remove('hidden');
                document.getElementById('revealSection').classList.add('hidden');
                document.getElementById('resultDisplay').classList.add('hidden');
                document.getElementById('fistDisplay').classList.add('hidden');
                
                generateMarbleGrid();
            }
        }
        
        function setupGuessButtons() {
            const container = document.getElementById('guessButtons');
            container.innerHTML = '';
            
            if (gameState.callMode === 'evenOdd') {
                container.innerHTML = `
                    <button class="guess-btn even" onclick="makeGuess('even')">EVEN</button>
                    <button class="guess-btn odd" onclick="makeGuess('odd')">ODD</button>
                `;
            } else {
                container.innerHTML = `
                    <button class="guess-btn kali" onclick="makeGuess('kali')">KALI (Odd)</button>
                    <button class="guess-btn jhota" onclick="makeGuess('jhota')">JHOTA (Even)</button>
                `;
            }
        }
        
        function updateGameUI() {
            document.getElementById('currentRound').textContent = gameState.currentRound;
            document.getElementById('player1Name').textContent = 'You';
            document.getElementById('player1Marbles').textContent = gameState.marbles;
            document.getElementById('player2Name').textContent = gameState.opponentName;
            document.getElementById('player2Marbles').textContent = gameState.opponentMarbles;
        }
        
        function generateMarbleGrid() {
            const grid = document.getElementById('marbleGrid');
            grid.innerHTML = '';
            gameState.selectedMarbles.clear();
            
            for (let i = 1; i <= 20; i++) {
                const marble = document.createElement('div');
                marble.className = 'marble-item';
                marble.dataset.marbleId = i;
                marble.onclick = () => toggleMarbleSelection(i);
                grid.appendChild(marble);
            }
            
            updateSelectedCount();
        }
        
        function toggleMarbleSelection(marbleId) {
            if (gameState.selectedMarbles.has(marbleId)) {
                gameState.selectedMarbles.delete(marbleId);
                document.querySelector(`[data-marble-id="${marbleId}"]`).classList.remove('selected');
            } else {
                if (gameState.selectedMarbles.size < 20) {
                    gameState.selectedMarbles.add(marbleId);
                    document.querySelector(`[data-marble-id="${marbleId}"]`).classList.add('selected');
                }
            }
            updateSelectedCount();
        }
        
        function updateSelectedCount() {
            document.getElementById('selectedCount').textContent = gameState.selectedMarbles.size;
        }
        
        function clearMarbleSelection() {
            gameState.selectedMarbles.clear();
            document.querySelectorAll('.marble-item').forEach(m => m.classList.remove('selected'));
            updateSelectedCount();
        }
        
        function confirmMarbleSelection() {
            if (gameState.selectedMarbles.size === 0) {
                showNotification('Select at least 1 marble!');
                return;
            }
            
            gameState.hiddenMarbles = gameState.selectedMarbles.size;
            gameState.opponentGuess = Math.random() > 0.5 ? 'even' : 'odd';
            
            document.getElementById('marbleSelectionSection').classList.add('hidden');
            document.getElementById('fistDisplay').classList.remove('hidden');
            document.getElementById('fistContainer').textContent = '✊';
            document.getElementById('fistContainer').className = 'fist-container fist-closed';
            document.getElementById('revealedMarbles').classList.add('hidden');
            document.getElementById('fistLabel').textContent = gameState.opponentName + ' is guessing...';
            
            setTimeout(() => {
                document.getElementById('fistDisplay').classList.add('hidden');
                document.getElementById('revealSection').classList.remove('hidden');
            }, 2000);
            
            showNotification(`You hid ${gameState.selectedMarbles.size} marbles`);
        }
        
        function makeGuess(guess) {
            const betAmount = parseInt(document.getElementById('betAmount').value);
            if (betAmount < 1 || betAmount > gameState.marbles) {
                showNotification('Invalid bet amount!');
                return;
            }
            
            const hiddenMarbles = Math.floor(Math.random() * 20) + 1;
            const isEven = hiddenMarbles % 2 === 0;
            
            let correctGuess = false;
            if (gameState.callMode === 'evenOdd') {
                correctGuess = (guess === 'even' && isEven) || (guess === 'odd' && !isEven);
            } else {
                correctGuess = (guess === 'jhota' && isEven) || (guess === 'kali' && !isEven);
            }
            
            document.getElementById('fistContainer').textContent = '✋';
            document.getElementById('fistContainer').className = 'fist-container fist-open';
            
            setTimeout(() => {
                showRevealedMarbles(hiddenMarbles);
                setTimeout(() => {
                    showResult(correctGuess, hiddenMarbles, betAmount);
                }, 1000);
            }, 500);
        }
        
        function showRevealedMarbles(count) {
            const container = document.getElementById('revealedMarbles');
            container.innerHTML = '';
            container.classList.remove('hidden');
            
            for (let i = 0; i < count; i++) {
                const marble = document.createElement('div');
                marble.className = 'revealed-marble';
                marble.style.animationDelay = (i * 0.05) + 's';
                container.appendChild(marble);
            }
        }
        
        function revealMarbles() {
            const isEven = gameState.hiddenMarbles % 2 === 0;
            const opponentCorrect = (gameState.opponentGuess === 'even' && isEven) || 
                                   (gameState.opponentGuess === 'odd' && !isEven);
            
            document.getElementById('revealSection').classList.add('hidden');
            document.getElementById('fistDisplay').classList.remove('hidden');
            document.getElementById('fistContainer').textContent = '✋';
            document.getElementById('fistContainer').className = 'fist-container fist-open';
            document.getElementById('fistLabel').textContent = `Opening fist... ${gameState.hiddenMarbles} marbles!`;
            
            showRevealedMarbles(gameState.hiddenMarbles);
            
            setTimeout(() => {
                showResult(!opponentCorrect, gameState.hiddenMarbles, 10);
            }, 1500);
        }
        
        function showResult(won, marbles, bet) {
            document.getElementById('fistDisplay').classList.add('hidden');
            document.getElementById('guessingSection').classList.add('hidden');
            document.getElementById('resultDisplay').classList.remove('hidden');
            
            const resultIcon = document.getElementById('resultIcon');
            const resultTitle = document.getElementById('resultTitle');
            const resultDetails = document.getElementById('resultDetails');
            
            resultTitle.classList.remove('win', 'lose');
            
            if (won) {
                gameState.marbles += bet;
                gameState.opponentMarbles -= bet;
                gameState.wins++;
                
                resultIcon.textContent = '🎉';
                resultTitle.textContent = 'You Won!';
                resultTitle.classList.add('win');
                resultDetails.textContent = `The marbles were ${marbles}. You earned ${bet} marbles!`;
                
                showMarbleChange(bet, true);
            } else {
                gameState.marbles -= bet;
                gameState.opponentMarbles += bet;
                
                resultIcon.textContent = '😔';
                resultTitle.textContent = 'You Lost!';
                resultTitle.classList.add('lose');
                resultDetails.textContent = `The marbles were ${marbles}. You lost ${bet} marbles.`;
                
                showMarbleChange(bet, false);
            }
            
            updateGameUI();
            updateUI();
            
            if (gameState.marbles <= 0) {
                setTimeout(() => showGameOver(), 2000);
            }
            
            saveGameData();
        }
        
        function showMarbleChange(amount, isGain) {
            const changeEl = document.getElementById('marbleChange');
            changeEl.textContent = (isGain ? '+' : '-') + amount;
            changeEl.className = 'marble-change ' + (isGain ? 'gain' : 'loss');
            void changeEl.offsetWidth;
            setTimeout(() => { changeEl.className = 'marble-change'; }, 2000);
        }
        
        function nextRound() {
            const resultTitle = document.getElementById('resultTitle');
            const wonLastRound = resultTitle.classList.contains('win');
            
            gameState.playerRole = wonLastRound ? 'hider' : 'guesser';
            gameState.currentRound++;
            setupRound();
        }
        
        function resetGame() {
            gameState.currentRound = 1;
            gameState.currentMode = null;
            gameState.playerRole = 'guesser';
            gameState.hiddenMarbles = 0;
            gameState.selectedMarbles.clear();
            gameState.opponentGuess = null;
            gameState.opponentMarbles = 100;
        }
        
        function showGameOver() {
            document.getElementById('adWatchCounter').textContent = `Ad watches remaining: ${gameState.adWatchesRemaining}/3`;
            const adBtn = document.getElementById('watchAdRevivalBtn');
            adBtn.disabled = gameState.adWatchesRemaining <= 0;
            adBtn.style.opacity = gameState.adWatchesRemaining <= 0 ? '0.5' : '1';
            document.getElementById('gameOverModal').classList.add('active');
        }
        
        function watchAdForRevival() {
            if (gameState.adWatchesRemaining <= 0) {
                showNotification('No ad watches remaining!');
                return;
            }
            
            document.getElementById('gameOverModal').classList.remove('active');
            showNotification('Watching ad...');
            setTimeout(() => {
                gameState.marbles += 50;
                gameState.adWatchesRemaining--;
                showMarbleChange(50, true);
                updateUI();
                saveGameData();
                showNotification(`+50 marbles! ${gameState.adWatchesRemaining} ad watches remaining`);
            }, 2000);
        }
        
        function buyMarblesForRevival() {
            document.getElementById('gameOverModal').classList.remove('active');
            showNotification('Processing purchase...');
            setTimeout(() => {
                gameState.marbles += 100;
                gameState.adWatchesRemaining = 3;
                showMarbleChange(100, true);
                updateUI();
                saveGameData();
                showNotification('Purchase successful! +100 marbles');
            }, 1500);
        }
        
        function openGameInfo() {
            document.getElementById('infoModal').classList.add('active');
        }
        
        function closeGameInfo() {
            document.getElementById('infoModal').classList.remove('active');
        }
        
        function openSettings() {
            document.getElementById('settingsModal').classList.add('active');
        }
        
        function closeSettings() {
            document.getElementById('settingsModal').classList.remove('active');
        }
        
        function toggleSound() {
            gameState.soundEnabled = !gameState.soundEnabled;
            document.getElementById('soundIndicator').textContent = gameState.soundEnabled ? '🔊' : '🔇';
            document.getElementById('soundStatus').textContent = gameState.soundEnabled ? 'ON' : 'OFF';
            saveGameData();
        }
        
        function toggleLanguage() {
            gameState.language = gameState.language === 'en' ? 'hi' : 'en';
            document.getElementById('langBtn').textContent = gameState.language === 'en' ? 'हिंदी में खेलें' : 'Play in English';
            saveGameData();
        }
        
        function updateUI() {
            document.getElementById('marbleCount').textContent = gameState.marbles;
            document.getElementById('totalWins').textContent = gameState.wins;
            document.getElementById('bonusPoints').textContent = gameState.bonusPoints;
        }
        
        function showNotification(message) {
            const notification = document.getElementById('notification');
            notification.textContent = message;
            notification.style.display = 'block';
            setTimeout(() => { notification.style.display = 'none'; }, 3000);
        }
        
        function saveGameData() {
            localStorage.setItem('kaliJhotaGame', JSON.stringify(gameState));
        }
        
        function loadGameData() {
            const saved = localStorage.getItem('kaliJhotaGame');
            if (saved) {
                const savedState = JSON.parse(saved);
                gameState = { ...gameState, ...savedState };
                gameState.currentRound = 1;
                gameState.currentMode = null;
                gameState.playerRole = 'guesser';
                gameState.hiddenMarbles = 0;
                gameState.selectedMarbles = new Set();
                gameState.opponentGuess = null;
            }
        }
        
        window.addEventListener('load', initGame);
        window.addEventListener('beforeunload', saveGameData);
